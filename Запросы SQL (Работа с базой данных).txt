Задание 1.
SELECT c.login, COUNT("inDelivery" = true) AS "inDelivery" FROM "Couriers" AS c INNER JOIN "Orders" AS o ON c.id=o."courierId"  GROUP BY c.login ORDER BY c.login;

SELECT *, CASE WHEN finished = true THEN 2 WHEN cancelled = true THEN -1 WHEN "inDelivery" = true THEN 1 ELSE 0 END FROM "Orders";
